Step 2: Initial Training (tune.py)

from pathlib import Path
from ultralytics import YOLO

if __name__ == "__main__":
    root = Path(__file__).resolve().parent
    data_yaml = (root / "data.yaml").resolve()

    if not data_yaml.exists():
        raise FileNotFoundError(f"data.yaml not found: {data_yaml}")

    print(f"Using data.yaml: {data_yaml}")
    print("Initializing yolo11s.pt (will auto-download if missing)...")

    model = YOLO("yolo11s.pt")

    results = model.train(
        data=str(data_yaml),
        epochs=200,
        imgsz=512,
        batch=16,
        device=0,
        workers=2,
        name="shark_yolo11s",
        patience=0,
        optimizer="AdamW",
        augment=True,
        amp=False,
        lr0=0.001,
        save_period=1
    )

    model.val()
    model.export(format="onnx")
    print("Best model saved at: runs/detect/shark_yolo11s/weights/best.onnx")

        resume=True,

Step 4: Build Web UI (app.py)

"""
Single-image shark polygon detection demo
"""
import gradio as gr
from ultralytics import YOLO

MODEL_PATH = "D:/new/runs/detect/shark_yolo11s2/weights/best.pt"

print("Loading model...")
model = YOLO(MODEL_PATH)

def shark_id(image):
    results = model.predict(image, conf=0.50, imgsz=640)
    annotated_img = results[0].plot()
    return annotated_img[:, :, ::-1]

demo = gr.Interface(
    fn=shark_id,
    inputs=gr.Image(type="numpy", label="Drop shark image here"),
    outputs=gr.Image(type="numpy", label="Detection Results"),
    title="YOLO11 Shark Identifier",
    description="Upload an image of a shark to test model detection accuracy."
)

if __name__ == "__main__":
    demo.launch(server_name="127.0.0.1", server_port=7860, share=True)

Kaggle Markdown Cell 2: Data Preparation Strategy

## Data Preparation Strategy

YOLO normally expects images in a single train/val folder. With multiple Kaggle inputs,
copying everything can waste too much storage and time.

Use text files that list absolute image paths. Then point data.yaml to those text files.
YOLO will map labels automatically.

Kaggle Code Cell 2: Dynamic Data Merging

import os
import yaml
from ultralytics import settings

def prepare_yolo_dataset():
    working_yaml = "/kaggle/working/kaggle_data.yaml"
    train_txt = "/kaggle/working/train_images.txt"
    val_txt = "/kaggle/working/val_images.txt"

    settings.update({'datasets_dir': '/kaggle/working'})

    train_dirs = [
        '/kaggle/input/datasets/shashikachamath/sharks/train/images',
        '/kaggle/input/datasets/shashikachamath/trainimagesbc'
    ]

    with open(train_txt, 'w') as f:
        for d in train_dirs:
            if os.path.exists(d):
                for img in os.listdir(d):
                    if img.lower().endswith(('.png', '.jpg', '.jpeg')):
                        f.write(os.path.join(d, img) + '\n')

    val_dirs = [
        '/kaggle/input/datasets/shashikachamath/sharks/valid/images',
        '/kaggle/input/datasets/shashikachamath/backgroundsval/val/images'
    ]

    with open(val_txt, 'w') as f:
        for d in val_dirs:
            if os.path.exists(d):
                for img in os.listdir(d):
                    if img.lower().endswith(('.png', '.jpg', '.jpeg')):
                        f.write(os.path.join(d, img) + '\n')

    yaml_data = {
        'path': '',
        'train': train_txt,
        'val': val_txt,
        'nc': 12,
        'names': {
            0: 'Class_0', 1: 'Class_1', 2: 'Class_2', 3: 'Class_3',
            4: 'Class_4', 5: 'Class_5', 6: 'Class_6', 7: 'Class_7',
            8: 'Class_8', 9: 'Class_9', 10: 'Class_10', 11: 'Class_11'
        }
    }

    with open(working_yaml, 'w') as f:
        yaml.dump(yaml_data, f)

    return working_yaml

yaml_path = prepare_yolo_dataset()
print(f"YAML generated at: {yaml_path}")

Kaggle Markdown Cell 3: Training the Model

## Training Phase

Now that data is mapped, start training with dual T4 GPUs using device=[0, 1].
Load a checkpoint to resume from where you left off.

Kaggle Code Cell 3: Execute Training

from ultralytics import YOLO

checkpoint_path = "/kaggle/input/models/shashikachamath/56/pytorch/default/1/epoch56.pt"
model = YOLO(checkpoint_path)

results = model.train(
    data="/kaggle/working/kaggle_data.yaml",
    epochs=46,
    imgsz=512,
    batch=32,
    device=[0, 1],
    optimizer="AdamW",
    lr0=0.001,
    amp=False,
    name="shark_dual_gpu_resumed",
    project="/kaggle/working/runs",
    exist_ok=True,
    save_period=2,
    workers=8
)

print("Training Complete!")
